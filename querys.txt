/******Create a script to list animals older than 2 years and female, sorted by name******/
USE [ANIMALES]
GO

SELECT [ANIMALID]
      ,[NAME]
      ,[BREED]
	   ,[BIRTHDATE]
      ,[SEX]
      ,[PRICE]
      ,[STATUS],
 DATEDIFF(YEAR, [BIRTHDATE], GETDATE()) as 'age'
 FROM [dbo].[ANIMALES]
 WHERE SEX= 'female'
 AND DATEDIFF(YEAR, [BIRTHDATE], GETDATE()) > 2
 ORDER BY [NAME];
GO
